import h5py
import numpy as np
import matplotlib.pyplot as plt

# Zayyay's Constant: The predicted floor
ZAYYAY_CONSTANT = 0.0041 

def check_zayyay_floor(file_path):
    """Loads GW posterior samples and checks the chi_eff distribution against the Zayyay floor."""
    
    print(f"--- Analyzing: {file_path} ---")
    
    try:
        # Load the HDF5 file
        with h5py.File(file_path, 'r') as f:
            # Common path for bilby/pesummary outputs
            if 'posterior_samples' in f and 'chi_eff' in f['posterior_samples']:
                chi_eff_samples = f['posterior_samples']['chi_eff'][:]
            elif 'chi_eff' in f['/']:
                 chi_eff_samples = f['chi_eff'][:]
            else:
                print("Error: 'chi_eff' parameter not found.")
                return

        abs_chi_eff = np.abs(chi_eff_samples)
        
        # Plot the distribution
        plt.figure(figsize=(8, 5))
        plt.hist(abs_chi_eff, bins=50, density=True, alpha=0.6, label='Posterior $P(|\chi_{\\text{eff}}|)$')
        
        # Plot the Zayyay floor
        plt.axvline(ZAYYAY_CONSTANT, color='red', linestyle='--', linewidth=2, 
                    label=f'Zayyay Floor ($\\Xi={ZAYYAY_CONSTANT}$)')
        
        # Zoom in on the critical region near zero
        plt.xlim(0, 0.05) 
        plt.title(f'Test of Zayyay Floor on Effective Spin Magnitude ({file_path.split("_")[-1].split(".")[0]})')
        plt.xlabel('$|\chi_{\\text{eff}}|$')
        plt.ylabel('Probability Density')
        plt.legend()
        plt.grid(True, alpha=0.3)
        plt.savefig(f"Zayyay_Floor_Test_{file_path.split('_')[-1].split('.')[0]}.png")
        plt.show()

        # Quantitative check
        samples_below_floor = np.sum(abs_chi_eff < ZAYYAY_CONSTANT)
        total_samples = len(abs_chi_eff)
        
        print(f"Total samples: {total_samples}")
        print(f"Samples below Zayyay Floor (0.0041): {samples_below_floor} ({(samples_below_floor/total_samples)*100:.3f}%)")
        
    except Exception as e:
        print(f"An error occurred: {e}")
