{\LARGE \textbf{\textcolor{DeepBlue}{FIELDS' MILLENNIUM FALCON}}}\\[-0.2em]
{\large \textit{Ledgered Navier--Stokes Program --- Launch Manifest (One Pager)}}\\[0.6em]

\noindent\textbf{Mission.}
Strengthen localized energy analysis for 3D incompressible NSE by \emph{accounting} for unresolved flux via a nonnegative defect/ledger measure, derive a scale-invariant $\varepsilon$-regularity criterion, and ship a conservative diagnostic that prints the energy ledger to truncation order.

\vspace{0.5em}
\noindent\textbf{Core Identity (Ledgered Local Energy; Leray--Hopf).}
\begin{equation*}
\partial_t\!\Big(\tfrac12|v|^2\Big)+\nabla\!\cdot\!\Big[\big(\tfrac12|v|^2+p\big)v\Big]
=\nu\,\Delta\!\Big(\tfrac12|v|^2\Big)-\nu|\nabla v|^2-\mu,\qquad \mu\ge 0.
\end{equation*}
Weak form places $\int\phi\,d\mu$ on the LHS as a positive term, sharpening Caccioppoli.

\vspace{0.4em}
\noindent\textbf{$\varepsilon$-Regularity (ledger saturation).}
For parabolic cylinders $Q_r$ with scaled CKNS energy $\mathcal E(r)$ and ledger mass $\Lambda(r)=r^{-1}\mu(Q_r)$, smallness of $\mathcal E(r_0)+\Lambda(r_0)$ implies H\"older regularity on $Q_{\delta r_0}$.

\vspace{0.5em}
\noindent\textbf{Evidence Engine (conservative diagnostic).}
2D vorticity--streamfunction NSE (periodic), Arakawa Jacobian + FFT Poisson:
\begin{equation*}
E(t)+\int_0^t \nu\|\omega\|_2^2 = E(0) + \mathcal O(\Delta t^2).
\end{equation*}
