\begin{block}{Motivation \& Problem}
\textbf{Goal.} Strengthen the localized energy framework for Leray--Hopf solutions of 3D NSE by \emph{accounting} for unresolved nonlinear flux via a nonnegative defect/ledger measure.\\[0.3em]
\textbf{Why it matters.} Places a positive term on the LHS of the local balance $\Rightarrow$ sharper Caccioppoli estimates and a scale-invariant $\varepsilon$-regularity criterion.
\end{block}

\begin{block}{Ledgered Local Energy Identity (Theorem 1)}
Let $v$ be a Leray--Hopf solution with pressure $p$. There exists a nonnegative Radon measure $\mu$ such that in distributions
\begin{equation*}
\partial_t\!\Big(\tfrac12|v|^2\Big)+\nabla\!\cdot\!\Big[\big(\tfrac12|v|^2+p\big)v\Big]
=\nu\,\Delta\!\Big(\tfrac12|v|^2\Big)-\nu|\nabla v|^2-\mu.
\end{equation*}
For $\phi\ge0$ compactly supported,
\begin{align*}
\int \tfrac12|v|^2 \phi &+ \nu\!\int |\nabla v|^2\phi + \int \phi\,d\mu \\
&= \int \Big(\tfrac12|v|^2+p\Big) v\!\cdot\!\nabla\phi + \tfrac12|v|^2(\partial_t\phi+\nu\Delta\phi).
\end{align*}
\textbf{Ledger.} $\mu$ is the Duchon--Robert defect; for smooth flows $\mu\equiv0$.
\end{block}

\begin{block}{$\varepsilon$-Regularity with Ledger Saturation (Theorem 2)}
On cylinders $Q_r$, define the CKNS energy $\mathcal E(r)$ and scaled ledger mass $\Lambda(r)=r^{-1}\mu(Q_r)$.\\[0.3em]
\textbf{Criterion.} If $\mathcal E(r_0)+\Lambda(r_0)\le \varepsilon_*$ then $v$ is H\"older on $Q_{\delta r_0}$ (universal $\varepsilon_*,\delta>0$).
\end{block}

\begin{block}{Methods (One-slide view)}
\textbf{Analysis.} Mollify, derive exact balance + commutator, pass to limit; identify $\mu\ge0$ with Duchon--Robert defect.\\
\textbf{Regularity.} Caccioppoli with $\int\phi\,d\mu$ on the LHS; CKNS blow-up/compactness $\Rightarrow$ ancient limit with $\mu=0$ is trivial.\\
\textbf{Numerics.} Skew-symmetric convection; spectral Poisson; RK2; ledger residual plotted and shown to be $\mathcal O(\Delta t^2)$.
\end{block}
