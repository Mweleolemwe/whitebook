\begin{abstract}
We prove a \emph{ledgered} local energy identity for Leray--Hopf solutions of the three-dimensional incompressible Navier--Stokes equations, augmenting the classical localized energy inequality by a nonnegative Radon measure $\mu$ that captures unresolved nonlinear flux (the \emph{defect/ledger measure}). In distributional form,
\[
\partial_t\!\Big(\tfrac12|v|^2\Big)+\nabla\!\cdot\!\Big[\big(\tfrac12|v|^2+p\big)v\Big]
=\nu\,\Delta\!\Big(\tfrac12|v|^2\Big)-\nu|\nabla v|^2-\mu,
\]
and, for nonnegative test functions, the corresponding weak formulation places $\int\phi\,d\mu$ on the left-hand side as a positive term, strengthening Caccioppoli-type estimates. As an application, we obtain a scale-invariant $\varepsilon$-regularity criterion on parabolic cylinders in terms of the Caffarelli--Kohn--Nirenberg energy $\mathcal E(r)$ \emph{plus} the scaled ledger mass $\Lambda(r)=r^{-1}\mu(Q_r)$: smallness of $\mathcal E(r_0)+\Lambda(r_0)$ implies H\"older regularity on a smaller cylinder. We complement the analysis with a conservative finite-volume diagnostic (Arakawa Jacobian + FFT Poisson solve) that enforces the discrete energy ledger on periodic domains and reports the truncation-order residual. Code, data, and figure artifacts supporting the evidence engine are archived under a permanent DOI: \href{https://doi.org/10.6084/m9.figshare.30520784}{10.6084/m9.figshare.30520784}.
\end{abstract}

\section{Setting and notation}
Let $v:\Omega\times(0,T)\to\mathbb{R}^3$ and $p:\Omega\times(0,T)\to\mathbb{R}$ solve
\begin{equation}\label{NSE}
\partial_t v + (v\!\cdot\!\nabla)v + \nabla p = \nu\Delta v,\qquad \diverg v=0,
\end{equation}
in the Leray--Hopf sense on $\Omega=\mathbb{R}^3$ or $\mathbb{T}^3$, with $v\in L^\infty_t L^2_x\cap L^2_t\dot H^1_x$ obeying the global energy inequality.
For a parabolic cylinder $Q_r(z_0)=B_r(x_0)\times(t_0-r^2,t_0)$ set
\[
\mathcal{E}(r) := r^{-1}\!\int_{Q_r}\Big(|\nabla v|^2 + |p-p_{Q_r}|^{3/2}\Big)\,dx\,dt,\qquad
\Lambda(r) := r^{-1}\,\mu(Q_r).
\]

\section{Defect/ledger measure}
\begin{definition}[Ledger/defect measure]\label{def:ledger}
Let $\rho_\varepsilon$ be a standard space--time mollifier and $v^\varepsilon=v\ast\rho_\varepsilon$. Define
\[
d\mu_\varepsilon := \Big[(v\otimes v)^\varepsilon - v^\varepsilon\!\otimes v^\varepsilon\Big]\!:\!\nabla v^\varepsilon\,dx\,dt.
\]
Along a sequence $\varepsilon\downarrow0$ we have $d\mu_\varepsilon\rightharpoonup d\mu$ in $\mathcal{M}_{\mathrm{loc}}(\Omega\times(0,T))$ with $d\mu\ge0$. This coincides with the Duchon--Robert defect.
\end{definition}

\section{Ledgered local energy identity}
\begin{theorem}[Leray--Hopf ledgered local energy]\label{thm:ledger}
There exists a nonnegative Radon measure $\mu$ such that in distributions
\begin{equation}\label{eq:DR}
\partial_t \Big(\tfrac12|v|^2\Big) + \nabla\!\cdot\!\Big[\Big(\tfrac12|v|^2+p\Big)v\Big]
= \nu\,\Delta\!\Big(\tfrac12|v|^2\Big) - \nu|\nabla v|^2 - \mu.
\end{equation}
Equivalently, for $\phi\in C_c^\infty(\Omega\times[0,T))$, $\phi\ge0$,
\begin{align}\label{eq:ledger-weak}
\int \tfrac12|v|^2 \phi(\cdot,t)\,dx &+ \nu\!\int_0^t\!\!\int |\nabla v|^2 \phi\,dx\,ds + \int_0^t\!\!\int \phi\,d\mu \nonumber\\
&= \int_0^t\!\!\int \Big(\tfrac12|v|^2+p\Big) v\!\cdot\!\nabla\phi + \tfrac12|v|^2(\partial_s\phi+\nu\Delta\phi)\,dx\,ds
 + \int \tfrac12|v|^2 \phi(\cdot,0)\,dx.
\end{align}
\end{theorem}

\begin{remark}
Dropping $\int\phi\,d\mu$ in \eqref{eq:ledger-weak} recovers the classical local energy inequality. For smooth solutions $\mu\equiv0$ and \eqref{eq:ledger-weak} holds with equality.
\end{remark}

\section{$\varepsilon$--regularity with ledger saturation}
\begin{theorem}[$\varepsilon$--regularity]\label{thm:ereg}
There exist universal constants $\varepsilon_*,\delta>0$ such that if for some $Q_{r_0}(z_0)$
\begin{equation}\label{eq:smallness}
\mathcal{E}(r_0) + \Lambda(r_0) \le \varepsilon_*,
\end{equation}
then $v$ is H\"older continuous on $Q_{\delta r_0}(z_0)$.
\end{theorem}
